class WeatherIntent extends Intent
{
    public function activators(): array
    {
        return [
            $this->exact('/weather'),
            $this->exact('Tell me the weather for today'),
        ];
    }

    public function run(ReceivedMessage $message): void
    {
        $this->sendMessage('Weather is nice today.');
    }
}
